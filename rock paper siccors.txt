  <script>
    const icons = { rock: '✊', paper: '✋', scissors: '✌️' };
    const choices = ['rock', 'paper', 'scissors'];
    const beats = { rock: 'scissors', paper: 'rock', scissors: 'paper' };
    let scores = { you: 0, draw: 0, cpu: 0 };
 
    function play(playerChoice) {
      const cpuChoice = choices[Math.floor(Math.random() * 3)];
      document.getElementById('icon-you').textContent = icons[playerChoice];
      document.getElementById('icon-cpu').textContent = icons[cpuChoice];
 
      const youSide = document.getElementById('side-you');
      const cpuSide = document.getElementById('side-cpu');
      youSide.className = 'side';
      cpuSide.className = 'side';
 
      const resultEl = document.getElementById('result-text');
 
      if (playerChoice === cpuChoice) {
        scores.draw++;
        resultEl.textContent = "It's a draw!";
        resultEl.className = 'result draw';
      } else if (beats[playerChoice] === cpuChoice) {
        scores.you++;
        resultEl.textContent = 'You win this round!';
        resultEl.className = 'result win';
        youSide.className = 'side win';
        cpuSide.className = 'side lose';
      } else {
        scores.cpu++;
        resultEl.textContent = 'Computer wins this round!';
        resultEl.className = 'result lose';
        cpuSide.className = 'side win';
        youSide.className = 'side lose';
      }
 
      document.getElementById('score-you').textContent = scores.you;
      document.getElementById('score-draw').textContent = scores.draw;
      document.getElementById('score-cpu').textContent = scores.cpu;
    }
 
    function resetGame() {
      scores = { you: 0, draw: 0, cpu: 0 };
      document.getElementById('score-you').textContent = '0';
      document.getElementById('score-draw').textContent = '0';
      document.getElementById('score-cpu').textContent = '0';
      document.getElementById('icon-you').textContent = '❓';
      document.getElementById('icon-cpu').textContent = '❓';
      document.getElementById('result-text').textContent = 'Pick your move!';
      document.getElementById('result-text').className = 'result';
      document.getElementById('side-you').className = 'side';
      document.getElementById('side-cpu').className = 'side';
    }
  </script>
